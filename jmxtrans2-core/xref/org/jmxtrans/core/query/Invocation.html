23  package org.jmxtrans.core.query;
24  
25  import org.jmxtrans.core.log.Logger;
26  import org.jmxtrans.core.log.LoggerFactory;
27  import org.jmxtrans.core.results.QueryResult;
28  import org.jmxtrans.utils.time.Clock;
29  
30  import javax.annotation.Nonnull;
31  import javax.annotation.concurrent.Immutable;
32  import javax.annotation.concurrent.ThreadSafe;
33  import javax.management.MBeanServer;
34  import javax.management.MalformedObjectNameException;
35  import javax.management.ObjectName;
36  import java.util.Arrays;
37  import java.util.Objects;
38  import java.util.Set;
39  import java.util.concurrent.BlockingQueue;
40  
41  import static java.util.Objects.hash;
42  
43  /**
44   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
45   */
46  @Immutable
47  @ThreadSafe
48  public class Invocation {
49  
50      @Nonnull
51      private final Logger logger = LoggerFactory.getLogger(getClass().getName());
52      @Nonnull
53      public final ObjectName objectName;
54      @Nonnull
55      public final String operationName;
56      @Nonnull
57      public final String resultAlias;
58      @Nonnull
59      private final Object[] params;
60      @Nonnull
61      private final String[] signature;
62      @Nonnull
63      private final Clock clock;
64  
65      public Invocation(
66              @Nonnull String objectName,
67              @Nonnull String operationName,
68              @Nonnull Object[] params,
69              @Nonnull String[] signature,
70              @Nonnull String resultAlias,
71              @Nonnull Clock clock) {
72          try {
73              this.objectName = new ObjectName(objectName);
74          } catch (MalformedObjectNameException e) {
75              throw new IllegalArgumentException("Invalid objectName '" + objectName + "'", e);
76          }
77          this.operationName = operationName;
78          this.params = params.clone();
79          this.signature = signature.clone();
80          this.resultAlias = resultAlias;
81          this.clock = clock;
82      }
83  
84      public void invoke(@Nonnull MBeanServer mbeanServer, @Nonnull BlockingQueue<QueryResult> resultQueue) {
85          Set<ObjectName> objectNames = mbeanServer.queryNames(objectName, null);
86          for (ObjectName on : objectNames) {
87              try {
88                  Object result = mbeanServer.invoke(on, operationName, params, signature);
89                  resultQueue.add(new QueryResult(resultAlias, result, clock.currentTimeMillis()));
90              } catch (Exception e) {
91                  logger.warn("Exception invoking " + on + "#" + operationName + "(" + Arrays.toString(params) + ")", e);
92              }
93          }
94      }
95  
96      @Override
97      public boolean equals(Object o) {
98          if (this == o) return true;
99          if (o == null || getClass() != o.getClass()) return false;
100 
101         Invocation that = (Invocation) o;
102         return Objects.equals(objectName, that.objectName)
103                 && Objects.equals(operationName, that.operationName)
104                 && Objects.deepEquals(params, that.params)
105                 && Objects.deepEquals(signature, that.signature)
106                 && Objects.equals(resultAlias, that.resultAlias);
107 
108     }
109 
110     @Override
111     public int hashCode() {
112         return hash(objectName, operationName, params, signature, resultAlias);
113     }
114 
115     @Override
116     public String toString() {
117         return "Invocation{" +
118                 "objectName=" + objectName +
119                 ", operationName='" + operationName + '\'' +
120                 ", resultAlias='" + resultAlias + '\'' +
121                 ", params=" + Arrays.toString(params) +
122                 ", signature=" + Arrays.toString(signature) +
123                 '}';
124     }
125 }
