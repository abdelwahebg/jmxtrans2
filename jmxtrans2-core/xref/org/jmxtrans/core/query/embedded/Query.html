55  /**
56   * Describe a JMX query on which metrics are collected.
57   *
58   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
59   * @author Jon Stevens
60   */
61  public class Query implements QueryMBean {
62  
63      @Nonnull
64      private final Logger logger = LoggerFactory.getLogger(getClass().getName());
65  
66      /**
67       * ObjectName of the Query MBean(s) to monitor, can contain
68       */
69      @Nonnull
70      private final ObjectName objectName;
71  
72      @Nullable
73      private final String resultAlias;
74      /**
75       * JMX attributes to collect. As an array for {@link javax.management.MBeanServer#getAttributes(javax.management.ObjectName, String[])}
76       */
77      @Nonnull
78      private final Map<String, QueryAttribute> attributesByName;
79      /**
80       * Copy of {@link #attributesByName}'s {@link java.util.Map#entrySet()} for performance optimization
81       */
82      @Nonnull
83      private final String[] attributeNames;
84  
85      @Nonnull
86      private final QueryMetrics metrics;
87  
88      /**
89       * {@link javax.management.ObjectName} of this {@link QueryMBean}
90       */
91      @Nonnull
92      private final ObjectName queryMbeanObjectName;
93  
94      @Nullable
95      private MBeanServer mbeanServer;
96  
97      private final int maxResults;
98      
99      private Query(@Nonnull ObjectName objectName,
100                   @Nullable String resultAlias,
101                   @Nonnull List<QueryAttribute> attributes,
102                   @Nonnull ObjectName queryMbeanObjectName,
103                   int maxResults,
104                   @Nonnull Clock clock) {
105         this.objectName = objectName;
106         this.resultAlias = resultAlias;
107         this.maxResults = maxResults;
108         this.attributesByName = new HashMap<>();
109         for (QueryAttribute attribute : attributes) {
110             attributesByName.put(attribute.getName(), attribute);
111         }
112         this.attributeNames = attributesByName.keySet().toArray(new String[0]);
113         this.queryMbeanObjectName = queryMbeanObjectName;
114         metrics = new QueryMetrics(clock);
115     }
116 
117     public Iterable<QueryResult> collectMetrics(@Nonnull MBeanServerConnection mbeanServer, @Nonnull ResultNameStrategy resultNameStrategy) throws IOException {
118         Collection<QueryResult> results = new ArrayList<>();
119         try (NanoChronometer chrono = metrics.collectionDurationChronometer()) {
120             /*
121              * Optimisation tip: no need to skip 'mbeanServer.queryNames()' if the ObjectName is not a pattern
122              * (i.e. not '*' or '?' wildcard) because the mbeanserver internally performs the check.
123              * Seen on com.sun.jmx.interceptor.DefaultMBeanServerInterceptor
124              */
125             Set<ObjectName> matchingObjectNames = mbeanServer.queryNames(this.objectName, null);
126             logger.debug(format("Query %s returned %s", objectName, matchingObjectNames));
127 
128             for (ObjectName matchingObjectName : matchingObjectNames) {
129                 try {
130                     AttributeList jmxAttributes = mbeanServer.getAttributes(matchingObjectName, this.attributeNames);
131                     logger.debug(format("Query %s returned %s", matchingObjectName, jmxAttributes));
132                     for (Attribute jmxAttribute : jmxAttributes.asList()) {
133                         attributesByName.get(jmxAttribute.getName()).collectMetrics(
134                                 matchingObjectName, jmxAttribute.getValue(), results, this, resultNameStrategy, maxResults);
135 
136                         // early return if we reach maxResults
137                         if (results.size() >= maxResults) return results;
138                     }
139                 } catch (Exception e) {
140                     logger.warn(format("Exception processing query %s", this), e);
141                 }
142             }
143             return results;
144         } finally {
145             metrics.incrementCollected(results.size());
146             metrics.incrementCollectionsCount();
147         }
148     }

150     @PostConstruct
151     public void start() throws Exception {
152         if (mbeanServer != null) {
153             mbeanServer.registerMBean(this, queryMbeanObjectName);
154         }
155     }
156 
157     @PreDestroy
158     public void stop() throws Exception {
159         if (mbeanServer != null) {
160             mbeanServer.unregisterMBean(queryMbeanObjectName);
161         }
162     }

164     @Override
165     @Nonnull
166     public ObjectName getObjectName() {
167         return objectName;
168     }
169 
170     @Nonnull
171     public Collection<QueryAttribute> getQueryAttributes() {
172         return attributesByName.values();
173     }
174 
175     @Override
176     @Nullable
177     public String getResultAlias() {
178         return resultAlias;
179     }
180 
181     @Nonnull
182     public ObjectName getQueryMbeanObjectName() {
183         return queryMbeanObjectName;
184     }
185 
186     public int getMaxResults() {
187         return maxResults;
188     }
189 
190     @Override
191     public boolean equals(Object o) {
192         if (this == o) return true;
193         if (o == null || getClass() != o.getClass()) return false;
194 
195         Query that = (Query) o;
196 
197         return Objects.equals(attributesByName, that.attributesByName)
198                 && Objects.equals(objectName, that.objectName)
199                 && Objects.equals(resultAlias, that.resultAlias);
200     }
201 
202     @Override
203     public int hashCode() {
204         return hash(objectName, resultAlias, attributesByName);
205     }
206 
207     @Override
208     @Nonnull
209     public String toString() {
210         return "Query{" +
211                 "objectName=" + objectName +
212                 ", resultAlias='" + resultAlias + '\'' +
213                 ", attributes=" + attributesByName.values() +
214                 '}';
215     }

242     @Nonnull
243     public static Builder builder() {
244         return new Builder();

247     public static final class Builder {
248         @Nonnull private static final AtomicInteger queryIdSequence = new AtomicInteger();
249 
250         @Nullable private ObjectName objectName;
251         @Nullable private String resultAlias;
252         @Nonnull private final List<QueryAttribute> attributes = new ArrayList<>();
253         @Nonnull private final Clock clock;
254         private int maxResults = 50;
255 
256         private Builder() {
257             this.clock = new SystemClock();
258         }
259 
260         @Nonnull public Builder withObjectName(@Nonnull String objectName) {
261             try {
262                 withObjectName(new ObjectName(objectName));
263                 return this;
264             } catch (MalformedObjectNameException e) {
265                 throw new RuntimeException("Object name [" + objectName + "] is not valid, cannot build query.");
266             }
267         }
268 
269         @Nonnull
270         public Builder withObjectName(@Nonnull ObjectName objectName) {
271             this.objectName = objectName;
272             return this;
273         }
274 
275         public Builder withResultAlias(@Nullable String resultAlias) {
276             this.resultAlias = resultAlias;
277             return this;
278         }
279         
280         public Builder withMaxResults(int maxResults) {
281             this.maxResults = maxResults;
282             return this;
283         }
284 
285         public Builder addAttribute(@Nonnull String attributeName) {
286             addAttribute(QueryAttribute.builder(attributeName).build());
287             return this;
288         }
289 
290         public Builder addAttribute(@Nonnull QueryAttribute attribute) {
291             attributes.add(attribute);
292             return this;
293         }
294 
295         public Builder addAttributes(@Nonnull Collection<QueryAttribute> attributes) {
296             this.attributes.addAll(attributes);
297             return this;
298         }
299 
300         @Nonnull
301         public Query build() {
302             try {
303                 if (objectName == null) {
304                     throw new RuntimeException("Cannot create query without an object name, please check your code");
305                 }
306                 return new Query(
307                         objectName,
308                         resultAlias,
309                         attributes,
310                         new ObjectName("org.jmxtrans.embedded:Type=Query,id=" + queryIdSequence.incrementAndGet()),
311                         maxResults,
312                         clock
313                 );
314             } catch (MalformedObjectNameException e) {
315                 throw new RuntimeException("Object name [" + objectName + "] is not valid, cannot expose MBean for this query.");
316             }
317         }
318     }
