46  /**
47   * Describe a JMX MBean attribute to collect and hold the attribute collection logic.
48   * <p/>
49   * Collected values are sent to a {@linkplain java.util.concurrent.BlockingQueue}
50   * for later export to the target monitoring systems
51   * (see {@link #collectMetrics(javax.management.ObjectName, Object, java.util.Collection, Query, ResultNameStrategy, int)}.
52   *
53   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
54   * @author Jon Stevens
55   */
56  public class QueryAttribute {
57  
58      @Nonnull
59      private final Logger logger = LoggerFactory.getLogger(getClass().getName());
60  
61      /**
62       * Name of the JMX Attribute to collect
63       */
64      @Nonnull
65      private final String name;
66      /**
67       * Used to create the name of the {@link org.jmxtrans.core.results.QueryResult} that will be exported.
68       * <p/>
69       * <code>null</code> if not defined in the configuration. The {@link #name} is then used to create to {@linkplain org.jmxtrans.core.results.QueryResult}
70       *
71       * @see org.jmxtrans.core.results.QueryResult#getName()
72       */
73      @Nullable
74      private final String resultAlias;
75  
76      /**
77       * Attribute type like '{@code gauge}' or '{@code counter}'. Used by monitoring systems like Librato who require this information.
78       *
79       * @see org.jmxtrans.core.results.QueryResult#getName()
80       */
81      @Nullable
82      private final String type;
83  
84      /**
85       * <code>null</code> if no 'key' as been defined in the config.
86       * Empty list if empty 'key' node has been declared in the config.
87       *
88       * @see javax.management.openmbean.CompositeType#keySet()
89       */
90      @Nullable
91      private final Set<String> keys;
92      
93      @Nonnull private final Clock clock;
94  
95      /**
96       * @param name        name of the JMX attribute
97       * @param type        type of the metric (e.g. "{@code counter}", "{@code gauge}", ...)
98       * @param resultAlias name of the result that will be exported
99       * @param keys        of the {@link javax.management.openmbean.CompositeData} to collect
100      * @param clock
101      */
102     private QueryAttribute(
103             @Nonnull String name,
104             @Nullable String type,
105             @Nullable String resultAlias,
106             @Nullable Set<String> keys,
107             @Nonnull Clock clock) {
108         this.name = Preconditions2.checkNotEmpty(name);
109         this.type = type;
110         this.resultAlias = resultAlias;
111         this.keys = keys;
112         this.clock = clock;
113     }
114 
115     @Nonnull
116     public String getName() {
117         return name;
118     }
119 
120     @Nullable
121     public String getResultAlias() {
122         return resultAlias;
123     }
124 
125     @Nullable
126     public String getType() {
127         return type;
128     }
129 
130     /**
131      * @param objectName    <code>objectName</code> on which the <code>attribute</code> was obtained.
132      * @param value         value of the given attribute. A 'simple' value (String, Number, Date)
133      *                      or a {@link javax.management.openmbean.CompositeData}
134      * @param results       queue to which the the computed result(s) must be added
135      * @param query
136      * @param resultNameStrategy
137      * @return collected metrics count
138      */
139     public void collectMetrics(
140             @Nonnull ObjectName objectName,
141             @Nonnull Object value,
142             @Nonnull Collection<QueryResult> results,
143             @Nonnull Query query,
144             @Nonnull ResultNameStrategy resultNameStrategy,
145             int maxResults) {
146         if (value instanceof CompositeData) {
147             CompositeData compositeData = (CompositeData) value;
148             collectCompositeData(objectName, results, query, resultNameStrategy, compositeData, maxResults);
149         } else if (value instanceof Number || value instanceof String || value instanceof Date) {
150             collectScalar(objectName, value, results, query, resultNameStrategy);
151         }
152         logger.info(format("Ignore non CompositeData attribute value %s:%s:%s=%s", query, objectName, this, value));
153     }
154 
155     private void collectScalar(
156             @Nonnull ObjectName objectName,
157             @Nullable Object value,
158             @Nonnull Collection<QueryResult> results,
159             @Nonnull Query query,
160             @Nonnull ResultNameStrategy resultNameStrategy) {
161         if (keys != null && logger.isInfoEnabled()) {
162             logger.info(format("Ignore keys configured for 'simple' jmx attribute. %s:%s:%s", query, objectName, this));
163         }
164         String resultName = resultNameStrategy.getResultName(query, objectName, this);
165         QueryResult result = new QueryResult(resultName, getType(), value, clock.currentTimeMillis());
166         logger.debug("Collect " + result);
167         results.add(result);
168     }
169 
170     private void collectCompositeData(
171             @Nonnull ObjectName objectName,
172             @Nonnull Collection<QueryResult> results,
173             @Nonnull Query query,
174             @Nonnull ResultNameStrategy resultNameStrategy,
175             @Nonnull CompositeData compositeData,
176             int maxResults) {
177         String[] keysToCollect;
178         if (keys == null) {
179             keysToCollect = compositeData.getCompositeType().keySet().toArray(new String[0]);
180             logger.info(format("No 'key' has been configured to collect data on this Composite attribute, collect all keys. %s:%s:%s", query, objectName, this));
181         } else {
182             keysToCollect = keys.toArray(new String[keys.size()]);
183         }
184         for (String key : keysToCollect) {
185             String resultName = resultNameStrategy.getResultName(query, objectName, this, key);
186             Object compositeValue = compositeData.get(key);
187             if (compositeValue instanceof Number || compositeValue instanceof String || compositeValue instanceof Date) {
188                 QueryResult result = new QueryResult(resultName, getType(), compositeValue, clock.currentTimeMillis());
189                 logger.debug("Collect " + result);
190                 results.add(result);
191 
192                 // early return if we reach maxResults
193                 if (results.size() >= maxResults) return;
194             } else {
195                 logger.debug(format("Skip non supported value %s:%s:%s:%s=%s", query, objectName, this, key, compositeValue));
196             }
197         }
198     }
199 
200     @Override
201     public boolean equals(Object o) {
202         if (this == o) return true;
203         if (o == null || getClass() != o.getClass()) return false;
204 
205         QueryAttribute that = (QueryAttribute) o;
206 
207         return Objects.equals(keys, that.keys)
208                 && Objects.equals(name, that.name)
209                 && Objects.equals(resultAlias, that.resultAlias)
210                 && Objects.equals(type, that.type);
211     }
212 
213     @Override
214     public int hashCode() {
215         return hash(name, resultAlias, type, keys);
216     }
217 
218     @Override
219     @Nonnull
220     public String toString() {
221         return "QueryAttribute{" +
222                 "name='" + getName() + '\'' +
223                 ", resultAlias='" + getResultAlias() + '\'' +
224                 ", keys=" + (keys == null ? null : Arrays.asList(keys)) +
225                 '}';
226     }
227 
228     @Nonnull
229     public static Builder builder(@Nonnull String name) {
230         return new Builder(name);
231     }
232 
233     public static final class Builder {
234         @Nonnull
235         private String name;
236         @Nullable
237         private String type;
238         @Nullable
239         private String resultAlias;
240         @Nullable
241         private Set<String> keys;
242 
243         private Builder(@Nonnull String name) {
244             this.name = name;
245         }
246 
247         public Builder withResultAlias(@Nullable String resultAlias) {
248             this.resultAlias = resultAlias;
249             return this;
250         }
251 
252         public Builder withType(@Nullable String type) {
253             this.type = type;
254             return this;
255         }
256 
257         public void addKey(String key) {
258             if (keys == null) keys = new HashSet<>();
259             keys.add(key);
260         }
261 
262         public Builder withKeys(@Nullable Collection<String> keys) {
263             if (keys != null) {
264                 this.keys = new HashSet<>();
265                 this.keys.addAll(keys);
266             } else {
267                 this.keys = null;
268             }
269             return this;
270         }
271 
272         @Nonnull
273         public QueryAttribute build() {
274             return new QueryAttribute(name, type, resultAlias, keys, new SystemClock());
275         }
276     }
