23  package org.jmxtrans.core.results;
24  
25  import java.util.concurrent.TimeUnit;
26  
27  import javax.annotation.Nonnull;
28  import javax.annotation.Nullable;
29  import javax.annotation.concurrent.Immutable;
30  import javax.annotation.concurrent.ThreadSafe;
31  
32  import org.jmxtrans.utils.Preconditions2;
33  
34  import lombok.EqualsAndHashCode;
35  import lombok.Getter;
36  import lombok.ToString;
37  
38  import static java.util.concurrent.TimeUnit.MILLISECONDS;
39  
40  /**
41   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
42   */
43  @Immutable
44  @ThreadSafe
45  @EqualsAndHashCode
46  @ToString
47  public class QueryResult {
48      @Nonnull @Getter private final String name;
49      private final long epochInMillis;
50      @Nullable @Getter private final Object value;
51      @Nonnull @Getter private final MetricType type;
52  
53      /**
54       * @param name          plain name of the metric (variables (e.g. <code>%my-jmx-attr%</code>) must have been resolved).
55       * @param type          type of the metric (e.g. "{@code counter}", "{@code gauge}", ...)
56       * @param value         value of the collected metric
57       * @param epochInMillis collect time in millis (see {@link System#currentTimeMillis()})
58       */
59      public QueryResult(@Nonnull String name, @Nonnull MetricType type, @Nullable Object value, long epochInMillis) {
60          this.name = Preconditions2.checkNotEmpty(name);
61          this.value = value;
62          this.epochInMillis = epochInMillis;
63          this.type = type;
64      }
65  
66      public long getEpoch(TimeUnit timeUnit) {
67          return timeUnit.convert(epochInMillis, MILLISECONDS);
68      }
69  
70  }
