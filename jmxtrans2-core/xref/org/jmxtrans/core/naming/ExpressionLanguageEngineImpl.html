36  /**
37   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
38   */
39  public class ExpressionLanguageEngineImpl implements ExpressionLanguageEngine {
40      protected final Logger logger = LoggerFactory.getLogger(getClass().getName());
41  
42      public ExpressionLanguageEngineImpl() {
43          try {
44              InetAddress localHost = InetAddress.getLocalHost();
45              String hostName = localHost.getHostName();
46              String reversedHostName = StringUtils2.reverseTokens(hostName, ".");
47              String canonicalHostName = localHost.getCanonicalHostName();
48              String reversedCanonicalHostName = StringUtils2.reverseTokens(canonicalHostName, ".");
49              String hostAddress = localHost.getHostAddress();
50  
51              registerExpressionEvaluator("hostname", hostName);
52              registerExpressionEvaluator("reversed_hostname", reversedHostName);
53              registerExpressionEvaluator("escaped_hostname", hostName.replaceAll("\\.", "_"));
54              registerExpressionEvaluator("canonical_hostname", canonicalHostName);
55              registerExpressionEvaluator("reversed_canonical_hostname", reversedCanonicalHostName);
56              registerExpressionEvaluator("escaped_canonical_hostname", canonicalHostName.replaceAll("\\.", "_"));
57              registerExpressionEvaluator("hostaddress", hostAddress);
58              registerExpressionEvaluator("escaped_hostaddress", hostAddress.replaceAll("\\.", "_"));
59          } catch (Exception e) {
60              logger.warn("Exception resolving localhost, expressions like #hostname#, #canonical_hostname# or #hostaddress# will not be available", e);
61          }
62      }
63  
64      /**
65       * Function based evaluators for expressions like '#hostname#' or '#hostname_canonical#'
66       */
67      @Nonnull
68      private Map<String, Callable<String>> expressionEvaluators = new HashMap<String, Callable<String>>();
69  
70      /**
71       * Replace all the '#' based keywords (e.g. <code>#hostname#</code>) by their value.
72       *
73       * @param expression the expression to resolve (e.g. <code>"servers.#hostname#."</code>)
74       * @return the resolved expression (e.g. <code>"servers.tomcat5"</code>)
75       */
76      @Nonnull
77      public String resolveExpression(@Nonnull String expression) {
78          StringBuilder result = new StringBuilder(expression.length());
79  
80          int position = 0;
81          while (position < expression.length()) {
82              char c = expression.charAt(position);
83              if (c == '#') {
84                  int beginningSeparatorPosition = position;
85                  int endingSeparatorPosition = expression.indexOf('#', beginningSeparatorPosition + 1);
86                  if (endingSeparatorPosition == -1) {
87                      throw new IllegalStateException("Invalid expression '" + expression + "', no ending '#' after beginning '#' at position " + beginningSeparatorPosition);
88                  }
89                  String key = expression.substring(beginningSeparatorPosition + 1, endingSeparatorPosition);
90                  Callable<String> expressionProcessor = expressionEvaluators.get(key);
91                  String value;
92                  if (expressionProcessor == null) {
93                      value = "#unsupported_expression#";
94                      logger.info("Unsupported expression '" + key + "'");
95                  } else {
96                      try {
97                          value = expressionProcessor.call();
98                      } catch (Exception e) {
99                          value = "#expression_error#";
100                         logger.warn("Error evaluating expression '" + key + "'", e);
101                     }
102                 }
103                 StringUtils2.appendEscapedNonAlphaNumericChars(value, false, result);
104                 position = endingSeparatorPosition + 1;
105 
106             } else {
107                 result.append(c);
108                 position++;
109             }
110         }
111         if (logger.isDebugEnabled()) logger.debug("resolveExpression(" + expression + "): " + result);
112         return result.toString();
113 
114     }

116     @Nonnull
117     public String resolveExpression(@Nonnull String expression, @Nonnull ObjectName exactObjectName) {
118 
119         StringBuilder result = new StringBuilder(expression.length());
120 
121         int position = 0;
122         while (position < expression.length()) {
123             char c = expression.charAt(position);
124             if (c == '%') {
125                 int beginningSeparatorPosition = position;
126                 int endingSeparatorPosition = expression.indexOf('%', beginningSeparatorPosition + 1);
127                 if (endingSeparatorPosition == -1) {
128                     throw new IllegalStateException("Invalid expression '" + expression + "', no ending '%' after beginning '%' at position " + beginningSeparatorPosition);
129                 }
130                 String key = expression.substring(beginningSeparatorPosition + 1, endingSeparatorPosition);
131                 String value = exactObjectName.getKeyProperty(key);
132                 if (value == null) {
133                     value = "null";
134                 }
135                 StringUtils2.appendEscapedNonAlphaNumericChars(value, result);
136                 position = endingSeparatorPosition + 1;
137             } else if (c == '#') {
138                 int beginningSeparatorPosition = position;
139                 int endingSeparatorPosition = expression.indexOf('#', beginningSeparatorPosition + 1);
140                 if (endingSeparatorPosition == -1) {
141                     throw new IllegalStateException("Invalid expression '" + expression + "', no ending '#' after beginning '#' at position " + beginningSeparatorPosition);
142                 }
143                 String key = expression.substring(beginningSeparatorPosition + 1, endingSeparatorPosition);
144                 Callable<String> expressionProcessor = expressionEvaluators.get(key);
145                 String value;
146                 if (expressionProcessor == null) {
147                     value = "#unsupported_expression#";
148                     logger.info("Unsupported expression '" + key + "'");
149                 } else {
150                     try {
151                         value = expressionProcessor.call();
152                     } catch (Exception e) {
153                         value = "#expression_error#";
154                         logger.warn("Error evaluating expression '" + key + "'", e);
155                     }
156                 }
157                 StringUtils2.appendEscapedNonAlphaNumericChars(value, false, result);
158                 position = endingSeparatorPosition + 1;
159 
160             } else {
161                 result.append(c);
162                 position++;
163             }
164         }
165         if (logger.isDebugEnabled())
166             logger.debug("resolveExpression(" + expression + ", " + exactObjectName + "): " + result);
167 
168         return result.toString();
169     }

171     /**
172      * Registers an expression evaluator with a static value.
173      */
174     public void registerExpressionEvaluator(String expression, Callable<String> evaluator) {
175         expressionEvaluators.put(expression, evaluator);
176     }
177 
178     /**
179      * Registers an expression evaluator with a static value.
180      */
181     public void registerExpressionEvaluator(String expression, String value) {
182         expressionEvaluators.put(expression, new StaticEvaluator(value));
183     }
184 
185     @Nonnull
186     public Map<String, Callable<String>> getExpressionEvaluators() {
187         return expressionEvaluators;
188     }
189 
190     public static class StaticEvaluator implements Callable<String> {
191         final String value;
192 
193         public StaticEvaluator(String value) {
194             this.value = value;
195         }
196 
197         @Override
198         public String call() throws Exception {
199             return value;
200         }
201 
202         @Override
203         public String toString() {
204             return "StaticStringCallable{" +
205                     "value='" + value + '\'' +
206                     '}';
207         }
208     }
