23  package org.jmxtrans.core.naming;
24  
25  import javax.annotation.Nonnull;
26  import javax.management.ObjectName;
27  
28  /**
29   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
30   */
31  public interface ExpressionLanguageEngine {
32      /**
33       * Replace all the '#' based keywords (e.g. <code>#hostname#</code>) by their value.
34       *
35       * @param expression the expression to resolve (e.g. <code>"servers.#hostname#."</code>)
36       * @return the resolved expression (e.g. <code>"servers.tomcat5"</code>)
37       */
38      @Nonnull
39      String resolveExpression(@Nonnull String expression);
40  
41      String resolveExpression(@Nonnull String expression, @Nonnull ObjectName exactObjectName);
42  }
