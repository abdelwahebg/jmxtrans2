23  package org.jmxtrans.core.scheduler;
24  
25  import org.jmxtrans.core.scheduler.NaiveScheduler;
26  import org.jmxtrans.core.scheduler.QueryGenerator;
27  import org.junit.After;
28  import org.junit.Before;
29  import org.junit.Test;
30  import org.mockito.Mockito;
31  
32  import java.util.concurrent.ExecutorService;
33  import java.util.concurrent.ScheduledExecutorService;
34  
35  import static java.util.concurrent.Executors.newSingleThreadExecutor;
36  import static java.util.concurrent.Executors.newSingleThreadScheduledExecutor;
37  import static org.assertj.core.api.Assertions.assertThat;
38  
39  public class NaiveSchedulerTest {
40  
41      private NaiveScheduler scheduler;
42      private ExecutorService queryExecutor;
43      private ExecutorService resultExecutor;
44      private ScheduledExecutorService queryTimer;
45  
46      @Before
47      public void createScheduler() {
48          queryExecutor = newSingleThreadExecutor();
49          resultExecutor = newSingleThreadExecutor();
50          queryTimer = newSingleThreadScheduledExecutor();
51          QueryGenerator queryGenerator = Mockito.mock(QueryGenerator.class);
52          scheduler = new NaiveScheduler(queryExecutor, resultExecutor, queryTimer, queryGenerator, 1);
53      }
54  
55      @Test(expected = IllegalStateException.class)
56      public void cannotStopServiceIfNotRunning() throws InterruptedException {
57          scheduler.stop();
58      }
59  
60      @Test(expected = IllegalStateException.class)
61      public void cannotStartServiceTwice() throws InterruptedException {
62          scheduler.start();
63          scheduler.start();
64      }
65  
66      @Test
67      public void nothingIsRunningAfterStop() throws InterruptedException {
68          scheduler.start();
69          scheduler.stop();
70          assertThat(queryExecutor.isShutdown()).isTrue();
71          assertThat(resultExecutor.isShutdown()).isTrue();
72          assertThat(queryTimer.isShutdown()).isTrue();
73      }
74  
75      @After
76      public void stopScheduler() {
77          try {
78              scheduler.stop();
79          } catch (InterruptedException ignore) {
80          } catch (IllegalStateException ignore) {
81          }
82      }
83  
84  }
