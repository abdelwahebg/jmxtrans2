45  public class QueryTest {
46  
47      private static final String MOCK_MBEAN_NAME = "test:type=Mock,name=mock";
48      private DummyResultNameStrategy resultNameStrategy;
49      private MBeanServer mbeanServer;
50      private Mock mock;
51  
52      @Before
53      public void before() throws Exception {
54          resultNameStrategy = new DummyResultNameStrategy();
55          mock = new Mock("PS Eden Space", 87359488L);
56          mbeanServer = ManagementFactory.getPlatformMBeanServer();
57          mbeanServer.registerMBean(mock, new ObjectName(MOCK_MBEAN_NAME));
58      }
59  
60      @After
61      public void after() throws Exception {
62          mbeanServer.unregisterMBean(new ObjectName(MOCK_MBEAN_NAME));
63      }
64  
65      @Test
66      public void basic_attribute_return_simple_result() throws Exception {
67          Query query = new Query(MOCK_MBEAN_NAME, "CollectionUsageThreshold", resultNameStrategy);
68          Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(1);
69  
70          query.collectMetrics(mbeanServer, results);
71  
72          assertThat(results).hasSize(1);
73          QueryResult queryResult = results.poll();
74          assertThat(queryResult).isNotNull();
75          assertThat(queryResult.getName()).isEqualTo("CollectionUsageThreshold");
76          assertThat(queryResult.getValue()).isInstanceOf(Number.class);
77      }
78  
79      @Test
80      public void result_name_are_processed() {
81          Query query = new Query(MOCK_MBEAN_NAME, "CollectionUsageThreshold", resultNameStrategy);
82          Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(1);
83  
84          query.collectMetrics(mbeanServer, results);
85  
86          assertThat(resultNameStrategy.hasBeenCalled()).isTrue();
87      }
88

103     @Test
104     public void non_indexed_list_attribute_return_simple_result() throws Exception {
105         Query query = new Query(MOCK_MBEAN_NAME, "IntegerList", resultNameStrategy);
106         Queue<QueryResult> results = new ArrayBlockingQueue<QueryResult>(mock.getIntegerList().size());
107 
108         query.collectMetrics(mbeanServer, results);
109 
110         for (int i = 0; i < mock.getIntegerList().size(); i++) {
111             final String name = "IntegerList_" + i;
112             Optional<QueryResult> result = FluentIterable.from(results).firstMatch(new Predicate<QueryResult>() {
113                 @Override
114                 public boolean apply(@Nullable QueryResult input) {
115                     return input != null && name.equals(input.getName());
116                 }
117             });
118 
119             org.assertj.guava.api.Assertions.assertThat(result).isPresent();
120             assertThat(result.get().getValue()).isInstanceOf(Number.class);
121         }
122     }
