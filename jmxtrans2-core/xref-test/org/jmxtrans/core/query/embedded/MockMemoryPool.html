23  package org.jmxtrans.core.query.embedded;
24  
25  import javax.management.openmbean.*;
26  
27  /**
28   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
29   * @see java.lang.management.MemoryPoolMXBean
30   */
31  public class MockMemoryPool implements MockMemoryPoolMBean {
32  
33      /**
34       * @see java.lang.management.MemoryUsage
35       */
36      private CompositeData usage;
37  
38      private String name;
39  
40      /**
41       * @param name      e.g. PS Eden Space
42       * @param committed e.g. 87359488L
43       */
44      public MockMemoryPool(String name, long committed) {
45          this.name = name;
46          try {
47              CompositeType type = new CompositeType(
48                      "java.lang.management.MemoryUsage", "My Type Desc",
49                      new String[]{"committed", "init", "max", "used"},
50                      new String[]{"committed", "init", "max", "used"},
51                      new OpenType[]{SimpleType.LONG, SimpleType.LONG, SimpleType.LONG, SimpleType.LONG});
52              usage = new CompositeDataSupport(
53                      type,
54                      new String[]{"committed", "init", "max", "used"},
55                      new Long[]{committed, 16318464L, 88997888L, 87359488L});
56          } catch (Exception e) {
57              throw new RuntimeException(e);
58          }
59      }
60  
61      @Override
62      public CompositeData getUsage() {
63          return usage;
64      }
65  
66      @Override
67      public long getCollectionUsageThreshold() {
68          return 0;
69      }
70  
71      @Override
72      public String getName() {
73          return name;
74      }
75  }
