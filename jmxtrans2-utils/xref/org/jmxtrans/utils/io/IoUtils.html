23  package org.jmxtrans.utils.io;
24  
25  import javax.annotation.Nonnull;
26  import java.io.File;
27  import java.io.FileOutputStream;
28  import java.io.IOException;
29  import java.io.InputStream;
30  import java.io.OutputStream;
31  import java.nio.file.Files;
32  import java.nio.file.Paths;
33  
34  public final class IoUtils {
35  
36      private static final int COPY_BUFFER_SIZE = 512;
37  
38      private IoUtils() {
39      }
40  
41      /**
42       * Simple implementation without chunking if the source file is big.
43       *
44       * @param source
45       * @param destination
46       * @throws java.io.IOException
47       */
48      public static void doCopySmallFile(@Nonnull File source, @Nonnull File destination, boolean append) throws IOException {
49          if (destination.exists() && destination.isDirectory()) {
50              throw new IOException("Can not copy file, destination is a directory: " + destination.getAbsolutePath());
51          } else if (!destination.exists()) {
52              boolean renamed = source.renameTo(destination);
53              if (renamed) return;
54          }
55  
56          FileOutputStream fos = null;
57          try {
58              fos = new FileOutputStream(destination, append);
59              if (append) {
60                  fos.write("\n".getBytes("UTF-8"));
61              }
62              fos.write(Files.readAllBytes(Paths.get(source.getAbsolutePath())));
63          } finally {
64              if (fos != null) {
65                  fos.close();
66              }
67          }
68      }
69  
70      public static void copy(@Nonnull InputStream in, @Nonnull OutputStream out) throws IOException {
71          byte[] buffer = new byte[COPY_BUFFER_SIZE];
72          int len;
73          while ((len = in.read(buffer)) != -1) {
74              out.write(buffer, 0, len);
75          }
76      }
77  }
