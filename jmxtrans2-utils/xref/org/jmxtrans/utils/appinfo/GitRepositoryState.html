23  package org.jmxtrans.utils.appinfo;
24  
25  import lombok.Getter;
26  
27  import java.io.IOException;
28  import java.io.InputStream;
29  import java.util.Properties;
30  
31  @Getter
32  public class GitRepositoryState {
33      private final String branch;
34      private final String tags;
35      private final String describe;
36      private final String describeShort;
37      private final String commitId;
38      private final String buildUserName;
39      private final String buildUserEmail;
40      private final String buildTime;
41      private final String commitUserName;
42      private final String commitUserEmail;
43      private final String commitMessageShort;
44      private final String commitMessageFull;
45      private final String commitTime;
46  
47      private GitRepositoryState(Properties properties) {
48          this.branch = properties.get("git.branch").toString();
49          this.tags = properties.get("git.tags").toString();
50          this.describe = properties.get("git.commit.id.describe").toString();
51          this.describeShort = properties.get("git.commit.id.describe-short").toString();
52          this.commitId = properties.get("git.commit.id").toString();
53          this.buildUserName = properties.get("git.build.user.name").toString();
54          this.buildUserEmail = properties.get("git.build.user.email").toString();
55          this.buildTime = properties.get("git.build.time").toString();
56          this.commitUserName = properties.get("git.commit.user.name").toString();
57          this.commitUserEmail = properties.get("git.commit.user.email").toString();
58          this.commitMessageShort = properties.get("git.commit.message.short").toString();
59          this.commitMessageFull = properties.get("git.commit.message.full").toString();
60          this.commitTime = properties.get("git.commit.time").toString();
61      }
62  
63      public static GitRepositoryState load() throws IOException {
64          try (InputStream in = GitRepositoryState.class.getClassLoader().getResourceAsStream("git.properties")) {
65              Properties properties = new Properties();
66              properties.load(in);
67              return new GitRepositoryState(properties);
68          }
69      }
70  }
