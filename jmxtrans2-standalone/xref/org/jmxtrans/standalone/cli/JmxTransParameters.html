23  package org.jmxtrans.standalone.cli;
24  
25  import com.beust.jcommander.Parameter;
26  
27  import java.io.File;
28  import java.util.List;
29  
30  public class JmxTransParameters {
31  
32      @Parameter(
33              names = { "--configFiles", "-c"},
34              description = "List of configuration files.",
35              validateValueWith = ExistingFileValidator.class)
36      private List<File> configFiles;
37  
38      @Parameter(
39              names = { "--configDir", "-d"},
40              description = "List of configuration directories, all files in those directories will be loaded.",
41              validateValueWith = ExistingDirectoryValidator.class)
42      private List<File> configDirs;
43  
44      @Parameter(
45              names = { "--ignoreParsingErrors", "-i"},
46              description = "Start JmxTrans even if errors are found in configuration files.")
47      private boolean ignoreParsingErrors = false;
48  
49      public List<File> getConfigFiles() {
50          return configFiles;
51      }
52  
53      public List<File> getConfigDirs() {
54          return configDirs;
55      }
56  
57      public boolean ignoreParsingErrors() {
58          return ignoreParsingErrors;
59      }
60  }
