23  package org.jmxtrans.spring;
24  
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.beans.factory.config.BeanDefinition;
28  import org.springframework.beans.factory.support.AbstractBeanDefinition;
29  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
30  import org.springframework.beans.factory.xml.AbstractSingleBeanDefinitionParser;
31  import org.springframework.beans.factory.xml.ParserContext;
32  import org.springframework.util.StringUtils;
33  import org.w3c.dom.Element;
34  import org.w3c.dom.Node;
35  import org.w3c.dom.NodeList;
36  
37  import java.util.ArrayList;
38  import java.util.List;
39  
40  /**
41   * {@link org.springframework.beans.factory.xml.BeanDefinitionParser} for an {@link org.jmxtrans.spring.EmbeddedJmxTransFactory}.
42   *
43   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
44   */
45  public class EmbeddedJmxTransBeanDefinitionParser extends AbstractSingleBeanDefinitionParser {
46  
47      private static final String CONFIGURATION_ATTRIBUTE = "configuration";
48      private static final String IGNORE_CONFIGURATION_NOT_FOUND_ATTRIBUTE = "ignore-configuration-not-found";
49  
50      private final Logger logger = LoggerFactory.getLogger(getClass());
51  
52      @Override
53      protected Class getBeanClass(Element element) {
54          return EmbeddedJmxTransFactory.class;
55      }
56  
57      @Override
58      protected String resolveId(Element element, AbstractBeanDefinition definition, ParserContext parserContext) {
59          String id = element.getAttribute(ID_ATTRIBUTE);
60          return (StringUtils.hasText(id) ? id : "jmxtrans");
61      }
62  
63      @Override
64      protected void doParse(Element element, ParserContext parserContext, BeanDefinitionBuilder builder) {
65          builder.setRole(BeanDefinition.ROLE_APPLICATION);
66          builder.getRawBeanDefinition().setSource(parserContext.extractSource(element));
67  
68          if (element.hasAttribute(IGNORE_CONFIGURATION_NOT_FOUND_ATTRIBUTE)) {
69              builder.addPropertyValue("ignoreConfigurationNotFound", element.getAttribute(IGNORE_CONFIGURATION_NOT_FOUND_ATTRIBUTE));
70          }
71          List<String> configurationUrls = new ArrayList<String>();
72          if (element.hasAttribute(CONFIGURATION_ATTRIBUTE)) {
73              String configurationUrl = element.getAttribute(CONFIGURATION_ATTRIBUTE);
74              logger.debug("Add configuration from attribute {}", configurationUrl);
75              configurationUrls.add(configurationUrl);
76          }
77  
78          NodeList configurationNodeList = element.getElementsByTagNameNS(element.getNamespaceURI(), CONFIGURATION_ATTRIBUTE);
79          for (int i = 0; i < configurationNodeList.getLength(); i++) {
80              Node node = configurationNodeList.item(i);
81              if (node instanceof Element) {
82                  String configurationUrl = node.getTextContent();
83                  logger.debug("Add configuration from attribute {}", configurationUrl);
84                  configurationUrls.add(configurationUrl);
85              } else {
86                  throw new RuntimeException("Invalid configuration child element " + node);
87              }
88  
89          }
90          builder.addPropertyValue("configurationUrls", configurationUrls);
91      }
92  }
