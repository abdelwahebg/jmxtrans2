65  @ThreadSafe
66  public class LibratoWriter implements OutputStreamBasedOutputWriter {
67  
68      @Nonnull private final Logger logger = LoggerFactory.getLogger(getClass().getName());
69  
70      @Nonnull private final ThreadLocal<ResultsClassifier> resultsClassifier = new ThreadLocal<ResultsClassifier>() {
71          @Override
72          protected ResultsClassifier initialValue() {
73              return new ResultsClassifier();
74          }
75      };
76      @Nonnull private final JsonFactory jsonFactory;
77      @Nullable private final String source;
78  
79      public LibratoWriter(@Nonnull JsonFactory jsonFactory, @Nullable String source) {
80          this.jsonFactory = jsonFactory;
81          this.source = source;
82      }
83  
84      @Override
85      public void beforeBatch(@Nonnull OutputStream out) throws IOException {
86          resultsClassifier.get().clear();
87      }
88  
89      @Override
90      public int write(@Nonnull OutputStream out, @Nonnull QueryResult result) throws IOException {
91          resultsClassifier.get().addResult(result);
92          return 0;
93      }
94  
95      @Override
96      public int afterBatch(@Nonnull OutputStream out) throws IOException {
97          try(JsonGenerator jsonGenerator = jsonFactory.createGenerator(out, UTF8)) {
98              int resultsWritten = 0;
99              jsonGenerator.writeStartObject();
100             resultsWritten += writeResultsAs("counters", jsonGenerator, resultsClassifier.get().getCounters());
101             resultsWritten += writeResultsAs("gauges", jsonGenerator, resultsClassifier.get().getGauges());
102             jsonGenerator.writeEndObject();
103             jsonGenerator.flush();
104             return resultsWritten;
105         } finally {
106             resultsClassifier.remove();
107         }
108     }
109 
110     private int writeResultsAs(String name, JsonGenerator jsonGenerator, Iterable<QueryResult> results) throws IOException {
111         int counter = 0;
112         jsonGenerator.writeArrayFieldStart(name);
113         for (QueryResult result : results) {
114             jsonGenerator.writeStartObject();
115             
116             jsonGenerator.writeStringField("name", result.getName());
117             
118             if (source != null && !source.isEmpty()) {
119                 jsonGenerator.writeStringField("source", source);
120             }
121             
122             jsonGenerator.writeNumberField("measure_time", result.getEpoch(SECONDS));
123             
124             if (result.getValue() instanceof Number) {
125                 writeNumberField(jsonGenerator, "value", (Number)result.getValue());
126             } else {
127                 logger.info(format("Value for result [%s] is not a number, cannot send it to Librato", result));
128             }
129             
130             jsonGenerator.writeEndObject();
131             counter++;
132         }
133         jsonGenerator.writeEndArray();
134         return counter;
135     }
136 
137     private void writeNumberField(JsonGenerator jsonGenerator, String name, Number value) throws IOException {
138         if (value instanceof Integer) {
139             jsonGenerator.writeNumberField(name, (Integer)value);
140         } else if (value instanceof Long) {
141             jsonGenerator.writeNumberField(name, (Long)value);
142         } else if (value instanceof Float) {
143             jsonGenerator.writeNumberField(name, (Float)value);
144         } else if (value instanceof Double) {
145             jsonGenerator.writeNumberField(name, (Double)value);
146         } else if (value instanceof AtomicInteger) {
147             jsonGenerator.writeNumberField(name, ((AtomicInteger)value).get());
148         } else if (value instanceof AtomicLong) {
149             jsonGenerator.writeNumberField(name, ((AtomicLong) value).get());
150         }
151     }

153     @NotThreadSafe
154     private static final class ResultsClassifier {
155         @Nonnull private final Logger logger = LoggerFactory.getLogger(getClass().getName());
156         
157         @Nonnull private final Queue<QueryResult> counters = new ArrayDeque<>();
158         @Nonnull private final Queue<QueryResult> gauges = new ArrayDeque<>();
159 
160         public void addResult(@Nonnull QueryResult result) {
161             if ("counter".equals(result.getType())) {
162                 counters.add(result);
163             } else if ("gauge".equals(result.getType())) {
164                 gauges.add(result);
165             } else if (result.getType() == null) {
166                 logger.info(format("Unspecified type for result [%s], export it as counter", result));
167                 counters.add(result);
168             } else {
169                 logger.info(format("Unsupported metric type [%s] for result [%s], export it as counter", result.getType(), result));
170                 counters.add(result);
171             }
172         }
173         
174         public Iterable<QueryResult> getCounters() {
175             return counters;
176         }
177         
178         public Iterable<QueryResult> getGauges() {
179             return gauges;
180         }
181         
182         public void clear() {
183             counters.clear();
184             gauges.clear();
185         }
186     }
187     
188     @ThreadSafe
189     public static final class Factory implements OutputWriterFactory<BatchingOutputWriter<HttpOutputWriter<LibratoWriter>>> {
190 
191         @Nonnull
192         @Override
193         public BatchingOutputWriter<HttpOutputWriter<LibratoWriter>> create(@Nonnull Map settings) {
194             int batchSize = getInt(settings, "batchSize", 100);
195             URL url = parseUrl(getString(settings, "libratoUrl", "https://metrics-api.librato.com/v1/metrics"));
196             int timeoutInMillis = getInt(settings, "timeoutInMillis", 1000);
197             Proxy proxy = getProxy(settings);
198             String source = "hostname"; // FIXME: correct handling of source should be implemented after #60 is done.
199             String username = getString(settings, "username", null);
200             String token = getString(settings, "token", null);
201             
202             HttpOutputWriter.Builder<LibratoWriter> httpOutputWriter =
203                     builder(url, loadAppInfo(), new LibratoWriter(new JsonFactory(), source))
204                             .withContentType("application/json; charset=utf-8")
205                             .withTimeout(timeoutInMillis, MILLISECONDS);
206             
207             if (username != null && !username.isEmpty()) {
208                 httpOutputWriter.withAuthentication(username, token);
209             }
210             if (proxy != null) {
211                 httpOutputWriter.withProxy(proxy);
212             }
213             
214             return new BatchingOutputWriter<>(batchSize, httpOutputWriter.build());
215         }
216 
217         @Nullable
218         private Proxy getProxy(@Nonnull Map<String, String> settings) {
219             String proxyHost = getString(settings, "proxyHost", null);
220             Integer proxyPort = getInt(settings, "proxyPort", 0);
221             
222             if (proxyHost == null || proxyHost.isEmpty()) return null;
223             
224             return new Proxy(HTTP, new InetSocketAddress(proxyHost, proxyPort));
225         }
226 
227         private AppInfo loadAppInfo() {
228             try {
229                 return AppInfo.load(LibratoWriter.class);
230             } catch (IOException e) {
231                 throw new IllegalStateException("Could not load AppInfo", e);
232             }
233         }
234 
235         private URL parseUrl(String urlString) {
236             try {
237                 return new URL(urlString);
238             } catch (MalformedURLException e) {
239                 throw new IllegalArgumentException(format("Malformed Librato URL [%s]", urlString), e);
240             }
241         }
242     }
