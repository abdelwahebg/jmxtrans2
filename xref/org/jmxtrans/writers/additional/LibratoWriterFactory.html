23  package org.jmxtrans.writers.additional;
24  
25  import java.io.IOException;
26  import java.net.InetSocketAddress;
27  import java.net.MalformedURLException;
28  import java.net.Proxy;
29  import java.net.URL;
30  import java.util.Map;
31  
32  import javax.annotation.Nonnull;
33  import javax.annotation.Nullable;
34  import javax.annotation.concurrent.ThreadSafe;
35  
36  import org.jmxtrans.core.output.OutputWriterFactory;
37  import org.jmxtrans.core.output.support.BatchingOutputWriter;
38  import org.jmxtrans.core.output.support.HttpOutputWriter;
39  import org.jmxtrans.utils.appinfo.AppInfo;
40  
41  import com.fasterxml.jackson.core.JsonFactory;
42  
43  import static java.lang.String.format;
44  import static java.net.Proxy.Type.HTTP;
45  import static java.util.concurrent.TimeUnit.MILLISECONDS;
46  
47  import static org.jmxtrans.core.output.support.HttpOutputWriter.builder;
48  import static org.jmxtrans.utils.ConfigurationUtils.getInt;
49  import static org.jmxtrans.utils.ConfigurationUtils.getString;
50  
51  @ThreadSafe
52  public final class LibratoWriterFactory implements OutputWriterFactory<BatchingOutputWriter<HttpOutputWriter<LibratoWriter>>> {
53  
54      @Nonnull
55      @Override
56      public BatchingOutputWriter<HttpOutputWriter<LibratoWriter>> create(@Nonnull Map<String, String> settings) {
57          int batchSize = getInt(settings, "batchSize", 100);
58          URL url = parseUrl(getString(settings, "libratoUrl", "https://metrics-api.librato.com/v1/metrics"));
59          int timeoutInMillis = getInt(settings, "timeoutInMillis", 1000);
60          Proxy proxy = getProxy(settings);
61          String source = "hostname"; // FIXME: correct handling of source should be implemented after #60 is done.
62          String username = getString(settings, "username", null);
63          String token = getString(settings, "token", null);
64  
65          HttpOutputWriter.Builder<LibratoWriter> httpOutputWriter =
66                  builder(url, loadAppInfo(), new LibratoWriter(new JsonFactory(), source))
67                          .withContentType("application/json; charset=utf-8")
68                          .withTimeout(timeoutInMillis, MILLISECONDS);
69  
70          if (username != null && !username.isEmpty()) {
71              httpOutputWriter.withAuthentication(username, token);
72          }
73          if (proxy != null) {
74              httpOutputWriter.withProxy(proxy);
75          }
76  
77          return new BatchingOutputWriter<>(batchSize, httpOutputWriter.build());
78      }
79  
80      @Nullable
81      private Proxy getProxy(@Nonnull Map<String, String> settings) {
82          String proxyHost = getString(settings, "proxyHost", null);
83          Integer proxyPort = getInt(settings, "proxyPort", 0);
84  
85          if (proxyHost == null || proxyHost.isEmpty()) return null;
86  
87          return new Proxy(HTTP, new InetSocketAddress(proxyHost, proxyPort));
88      }
89  
90      private AppInfo loadAppInfo() {
91          try {
92              return AppInfo.load(LibratoWriter.class);
93          } catch (IOException e) {
94              throw new IllegalStateException("Could not load AppInfo", e);
95          }
96      }
97  
98      private URL parseUrl(String urlString) {
99          try {
100             return new URL(urlString);
101         } catch (MalformedURLException e) {
102             throw new IllegalArgumentException(format("Malformed Librato URL [%s]", urlString), e);
103         }
104     }
105 }
