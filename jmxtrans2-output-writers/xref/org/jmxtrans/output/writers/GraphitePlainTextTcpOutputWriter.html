33  import javax.annotation.Nonnull;
34  import javax.annotation.concurrent.NotThreadSafe;
35  import java.io.IOException;
36  import java.io.OutputStreamWriter;
37  import java.io.Writer;
38  import java.net.ConnectException;
39  import java.net.InetAddress;
40  import java.net.InetSocketAddress;
41  import java.net.Socket;
42  import java.net.UnknownHostException;
43  import java.nio.charset.Charset;
44  import java.util.Map;
45  
46  import static java.util.concurrent.TimeUnit.MILLISECONDS;
47  import static java.util.concurrent.TimeUnit.SECONDS;
48  import static org.jmxtrans.utils.ConfigurationUtils.getInt;
49  import static org.jmxtrans.utils.ConfigurationUtils.getString;

51  /**
52   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
53   */
54  @NotThreadSafe
55  public class GraphitePlainTextTcpOutputWriter implements OutputWriter {
56  
57      private final Logger logger = LoggerFactory.getLogger(getClass().getName());
58  
59      @Nonnull private final static Charset UTF_8 = Charset.forName("UTF-8");
60      protected String metricPathPrefix;
61      private Socket socket;
62      private Writer writer;
63      private final int socketConnectTimeoutInMillis;
64      @Nonnull private final InetSocketAddress serverAddress;
65      @Nonnull private final Clock clock = new SystemClock();
66  
67      protected GraphitePlainTextTcpOutputWriter(
68              String metricPathPrefix,
69              int socketConnectTimeoutInMillis,
70              @Nonnull InetSocketAddress serverAddress) {
71          this.metricPathPrefix = metricPathPrefix;
72          this.socketConnectTimeoutInMillis = socketConnectTimeoutInMillis;
73          this.serverAddress = serverAddress;
74  
75          logger.info("GraphitePlainTextTcpOutputWriter is configured with " + serverAddress + ", metricPathPrefix=" + metricPathPrefix +
76                  ", socketConnectTimeoutInMillis=" + socketConnectTimeoutInMillis);
77      }
78  
79      /**
80       * {@link java.net.InetAddress#getLocalHost()} may not be known at JVM startup when the process is launched as a Linux service.
81       *
82       * @return
83       */
84      @Nonnull
85      protected String buildMetricPathPrefix() {
86          if (metricPathPrefix != null) {
87              return metricPathPrefix;
88          }
89          String hostname;
90          try {
91              hostname = InetAddress.getLocalHost().getHostName().replaceAll("\\.", "_");
92          } catch (UnknownHostException e) {
93              hostname = "#unknown#";
94          }
95          metricPathPrefix = "servers." + hostname + ".";
96          return metricPathPrefix;
97      }
98  
99      @Override
100     public void write(@Nonnull QueryResult result) throws IOException {
101         String msg = buildMetricPathPrefix() + result.getName() + " " + result.getValue() + " " + SECONDS.convert(clock.currentTimeMillis(), MILLISECONDS);
102         try {
103             ensureGraphiteConnection();
104             logger.debug("Send '" + msg + "' to " + serverAddress);
105             writer.write(msg + "\n");
106         } catch (IOException e) {
107             logger.warn("Exception sending '" + msg + "' to " + serverAddress, e);
108             releaseGraphiteConnection();
109             throw e;
110         }
111     }
112 
113     private void releaseGraphiteConnection() {
114         if (writer != null) {
115             try {
116                 writer.close();
117             } catch (IOException e) {
118                 logger.warn("Exception closing writer for socket " + socket, e);
119             }
120             writer = null;
121         }
122         if (socket != null) {
123             try {
124                 socket.close();
125             } catch (IOException e) {
126                 logger.warn("Exception closing socket " + socket, e);
127             }
128             socket = null;
129         }
130     }
131 
132     private void ensureGraphiteConnection() throws IOException {
133         boolean socketIsValid;
134         try {
135             socketIsValid = socket != null &&
136                     socket.isConnected()
137                     && socket.isBound()
138                     && !socket.isClosed()
139                     && !socket.isInputShutdown()
140                     && !socket.isOutputShutdown();
141         } catch (Exception e) {
142             socketIsValid = false;
143         }
144         if (!socketIsValid) {
145             writer = null;
146             try {
147                 socket = new Socket();
148                 socket.setKeepAlive(true);
149                 socket.connect(
150                         serverAddress,
151                         socketConnectTimeoutInMillis);
152             } catch (IOException e) {
153                 ConnectException ce = new ConnectException("Exception connecting to " + serverAddress);
154                 ce.initCause(e);
155                 throw ce;
156             }
157         }
158         if (writer == null) {
159             writer = new OutputStreamWriter(socket.getOutputStream(), UTF_8);
160         }
161     }
162 
163     @Override
164     @Nonnull
165     public String toString() {
166         return "GraphitePlainTextTcpOutputWriter{" +
167                 ", " + serverAddress +
168                 ", metricPathPrefix='" + metricPathPrefix + '\'' +
169                 '}';
170     }
171 
172     public static final class Factory implements OutputWriterFactory<GraphitePlainTextTcpOutputWriter> {
173         public final static String SETTING_HOST = "host";
174         public final static String SETTING_PORT = "port";
175         public static final int SETTING_PORT_DEFAULT_VALUE = 2003;
176         public final static String SETTING_NAME_PREFIX = "namePrefix";
177         public final static String SETTING_SOCKET_CONNECT_TIMEOUT_IN_MILLIS = "socket.connectTimeoutInMillis";
178         public final static int SETTING_SOCKET_CONNECT_TIMEOUT_IN_MILLIS_DEFAULT_VALUE = 500;
179 
180         @Override
181         public GraphitePlainTextTcpOutputWriter create(Map<String, String> settings) {
182             String metricPathPrefix = getString(settings, SETTING_NAME_PREFIX, null);
183             int socketConnectTimeoutInMillis = getInt(settings,
184                     SETTING_SOCKET_CONNECT_TIMEOUT_IN_MILLIS,
185                     SETTING_SOCKET_CONNECT_TIMEOUT_IN_MILLIS_DEFAULT_VALUE);
186             InetSocketAddress graphiteAddress = new InetSocketAddress(
187                     getString(settings, SETTING_HOST),
188                     getInt(settings, SETTING_PORT, SETTING_PORT_DEFAULT_VALUE));
189 
190             return new GraphitePlainTextTcpOutputWriter(
191                     metricPathPrefix,
192                     socketConnectTimeoutInMillis,
193                     graphiteAddress);
194         }
195     }
